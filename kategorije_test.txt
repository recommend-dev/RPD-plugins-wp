//Product Cat Edit page
function rcmnd_taxonomy_edit_meta_field($term) {

    //getting term ID
    $term_id = $term->term_id;

    // retrieve the existing value(s) for this meta field.
    $rcmnd_category_id = get_term_meta($term_id, 'rcmnd_category_id', true);
    ?>
    <tr class="form-field">
        <th scope="row" valign="top"><label for="rcmnd_category_id"><?php _e('Recommend Category', 'rcmnd'); ?></label></th>
        <td>
            <input type="text" name="rcmnd_category_id" id="rcmnd_category_id" value="<?php echo esc_attr($rcmnd_category_id) ? esc_attr($rcmnd_category_id) : '';?>">
            <p class="text"><?php _e('Enter connected recommend category identificator value here. Find more on https://about.recommend.co/product-feed/', 'rcmnd'); ?></p>
        </td>
    </tr>
    <?php
}


add_action('edited_product_cat', 'rcmnd_save_taxonomy_custom_meta', 10, 2);
add_action('create_product_cat', 'rcmnd_save_taxonomy_custom_meta', 10, 2);

// Save extra taxonomy fields callback function.
function rcmnd_save_taxonomy_custom_meta($term_id) {
    $rcmnd_category_id = filter_input(INPUT_POST, 'rcmnd_category_id');
    update_term_meta($term_id, 'rcmnd_category_id', $rcmnd_category_id);
}


add_filter('manage_edit-product_cat_columns', 'rcmnd_customFieldsListTitle' ); //Register Function
add_action('manage_product_cat_custom_column', 'rcmnd_customFieldsListDisplay' , 10, 3); //Populating the Columns

/**
 * Meta Title and Description column added to category admin screen.
 *
 * @param mixed $columns
 * @return array
 */
function rcmnd_customFieldsListTitle( $columns ) {
    $columns['rcmnd_category_id'] = __( 'Recommend Category Id', 'woocommerce' );
    return $columns;
}
/**
 * Meta Title and Description column value added to product category admin screen.
 *
 * @param string $columns
 * @param string $column
 * @param int $id term ID
 *
 * @return string
 */
function rcmnd_customFieldsListDisplay( $columns, $column, $id ) {
    if ( 'rcmnd_category_id' == $column ) {
        $columns = esc_html( get_term_meta($id, 'rcmnd_category_id', true) );
    }

    return $columns;
}

add_shortcode( 'rcmnd_alg_product_category_id', 'rcmnd_alg_product_category_id' );
if ( ! function_exists( 'rcmnd_alg_product_category_id' ) ) {
    /*
     * rcmnd_alg_product_category_id.
     */
	function rcmnd_alg_product_category_id( $atts, $content = '' ) {
		global $product;
		if ( $product ){
				$category = reset(get_the_terms( $product->ID, 'product_cat' ));
			
				$rcmnd_category_id = get_term_meta($category->term_id, 'rcmnd_category_id', true);
				if ( $rcmnd_category_id ) {
					return $rcmnd_category_id;
				}
				return '';
			}
			
			return '';
    }
